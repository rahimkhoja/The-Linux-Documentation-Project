#A - INTRODUCAO

# Devido ao grande numero de pessoas que possuem a placa Diamond Viper V330
e que utilizam
# o servidor XFREE em baixa resolucao de cores, resolvi escrever esse HT.
# Descreverei duas maneiras de configurar o servidor; - editando o arquivo
XF86config e
# com a utilizacao do programa XF86Setup.

# B - REQUISITOS

# Este howto foi escrito S-O-M-E-N-T-E para a utilizacao da Viper v330,
portanto se
# voce nao tem esta placa, sera como atravessar o Mar Negro de Submarino!!!
# Esteja familiar com o ambiente em q está trabalhando e seu editor de
textos
# de sua preferencia. Imagino que o leitor edite o Xf86Config no proprio
# linux, com o editor pico ou o joe, que sao editores simples, uteis e muito
faceis de usar.
# Nao se esqueca que Linux, Unix, e variantes diferenciam letras maiusculas
de
# minusculas. A letra "Q" e diferente da letra"q".

# C - CONFIGURANDO

# Antes de escrever esse HT eu procurei em diversos sites pela documentacao
sobre a placa,
# contatei desenvolvedores, membros da Diamond,
# mas nenhuma documentacao passou as informacoes corretas. Nos proprios HTs
do servidor
# XFree, ha um equivoco que favorece confusao em muitas pessoas. O que
ocorre e uma
# citacao em uma tabela que exibe a placa/chipset e seu servidor
correspondente.

# Server       Chipset

# P9000        Diamond Viper (exceto a 9100) e outras
# SVGA         Trident 8900 & 9400, Cirrus Logic, C & T, ET400, outros

# Provavelmente seja um erro de interpretacao ou de grafia, pois o servidor
P9000
# suporta a Viper PCI e AGP de 2 Mb (nao sei se mais alguma) e NAO a v330. O
# servidor correto para a V330 é o SVGA.


# C.1 - CONFIGURANDO UTILIZANDO O XF86Setup

# Primeiramente, voce tera que te-lo instalado, e obvio  :-P
# Normalmente ele se localiza no diretorio /usr/X11R6/bin/XFSetup ou
# no diretorio /urr/bin/X11/XF86Setup
# Digite XF86Setup no shel correspondente
# Aparecera uma caixa de dialogo perguntando se voce quer usar a
# configuracao de video ja gravada. Selecione NOT e tecle <ENTER> ou <CR>
# Pressione <ENTER> novamente na proxima caixa de dialogo, para mudar o modo
grafico
# Depois de alguns segundos, aparecera o menu de configuracao. Se possuir
# mouse, clique no botao 'CARD', senao navegue com a tecla <TAB> ate
# selecionar o botao. Assim que selecionar, tecle <ENTER>. Sera mostrado
# diversas placas. Selecione a placa (card) Riva 128.
# Agora entre em Detailed Setup (botao localizado na parte inferior esquerdo
# do seu monitor) para selecionar o chipset. Escolha Riva 128.
# Mas voce precisa configurar um monitor, certo? Para tanto, voce deve saber
a
# frequencia vertical, horizontal (em hertz) e sua definiçao maxima (em
pixels).
# Procure no manual do monitor, no site do fabricante, apenas em ultimo caso
# use o que estiver mencionado atras do monitor (se estiver escrito). Muito
cuidado com
# a configuracao do monitor. Dependendo da qualidade do seu equipamento, um
# simples erro pode condena-lo.
# Escolha a configuracao correspondente ao seu monitor que se adapte com a
# resolucao que voce usa. As caixas-combo localizados na parte superior,
indicam a frequencia horizontal
# e vertical. O centro da tela indica configurações padronizadas (mais
utilizadas).
# Em seguida, entre em Modeselection (botao localizado na parte superior do
monitor).
# Agora selecione a resolucao (em pixels) e a definicao de cores que voce
quer usar.
# Se seu monitor é SVGA, recomendo usar 16bpp.
# Finalmente clique no botao DONE. O servidor ira agora testar sua
configuracao.
# Se aparecer uma frase semelhante a "_X11TransSocketUNIXConect" é porque a
configuracao
# esta errada. Agora, se aparecer uma tela branca, clique em SAVE.

# pronto, agora basta digitar startx


# C.1 - CONFIGURANDO MANUALMENTE

# Antes de mais nada, faca uma copia de seguranca do arquivo de
configuracao:

# cp /etc/XF86Config /etc/XF86Config.backup

# Se voce nao sabe como utilizar qualquer editor de textos em linux, citarei
alguns comando a seguir:
# Supondo que o arquivo XF86config esteja no diretorio /etc (como padrao),
# que esteja em seu diretorio raiz e que seja o super-usuario (root) ou
algum usuario privilegiado.
# Se nao souber o que estou falando, contate seu administrador de sistemas.

# joe /etc/Xf86config   => aparecera o conteudo do arquivo. Navegue por ele
utilizando as setas de direcao.

# Para salvar o arquivo sem sair do editor :  ^K-D
# Para salvar o arquivo e sair do editor   :  ^K-X
# Para sair sem alterar os dados do arquivo:  ^C

# Se o leitor esta se perguntando: ^K-D, que diabos e isso? Facil...
mantenha a tecla
# CTRL ou CONTROL pressionada enquanto pressiona a tecla K e a tecla D.


# Este arquivo e muito delicado, tenha tenha certeza do que esta fazendo!!!

# Abaixo vou exibir um exemplo do arquivo XF86config, faca as mudancas
# que forem necessarias para o teclado, mouse e monitor. A placa ja esta
configurada.

# Tudo que estiver apos o caracter # e tido como comentario. Ou seja, nao
# interfere na configuracao. Utilize este caracter se precisar comentar
algo.

# Section Files diz ao programa onde se localizam os arquivos requeridos
# pelo servidor XFree
# Recomendo manter todas essa mesma configuracao para uma alta performance.
# No final deste HT, direi onde conseguir os servidores

Section "Files"
   RgbPath    "/usr/X11R6/lib/X11/rgb"
   FontPath   "/usr/X11R6/lib/X11/fonts/local"
   FontPath   "/usr/X11R6/lib/X11/fonts/TrueType"
   FontPath   "/usr/X11R6/lib/X11/fonts/misc:unscaled"
   FontPath   "/usr/X11R6/lib/X11/fonts/75dpi:unscaled"
   FontPath   "/usr/X11R6/lib/X11/fonts/100dpi:unscaled"
   FontPath   "/usr/X11R6/lib/X11/fonts/Type1"
   FontPath   "/usr/X11R6/lib/X11/fonts/Speedo"
   FontPath   "/usr/X11R6/lib/X11/fonts/misc"
   FontPath   "/usr/X11R6/lib/X11/fonts/75dpi"
   FontPath   "/usr/X11R6/lib/X11/fonts/100dpi"
EndSection

Section "ServerFlags"
EndSection

# Esta secao descrece as configuracoes de teclado
# Se voce tem um Microsoft Natural Keyboard, voce pode usar use:
#    XkbModel    "microsoft"
# Para modificar a linguagem
# Por exemplo, o layaut brasileiro obtem-se assim:F
#    XkbLayout   "br"

Section "Keyboard"
   Protocol        "Standard"
   XkbRules        "xfree86"
   XkbModel        "abnt2"     # => tipo de teclado
   XkbLayout       "br"        # => Layout do teclado (idioma)
EndSection

Section "Pointer"
   Protocol        "PS/2"       # => Tipo do mouse
   Device          "/dev/mouse" # => A porta em que o mouse esta conectado
#   Emulate3Buttons     #se seu mouse tem apenas 2 botoes e vc quiser fazer
a emulacao,
#                        descomente  a linha anterior
EndSection

Section "Monitor"
   Identifier      "Primary Monitor"  #Mantenha como esta
   VendorName      "Unknown"          #Mantenha como esta
   ModelName       "Unknown"          #Mantenha como esta
   HorizSync       31.5-57.0          #Configuracao horizontal do seu
monitor |min-max
   VertRefresh     50-90              #Configuracao verticao do seu monitor
|min.-max.
   Modeline  "1024x768"   75.00 1024 1048 1184 1328 768 771 777
806 -hsync -vsync
EndSection

#Modeline diz ao servidor qual resolucao vc quer usar (800X600, 640X480, ou
outra qualquer)

#    ModeLine "1024x768" 45 1024 1048 1208 1264 768 776 784 817 Interlace

#    Mode "1024x768i"
#        DotClock 45
#        HTimings 1024 1048 1208 1264
#        VTimings 768 776 784 817
#        Flags  "Interlace"
#    EndMode

   Chipset         "riva128"          #Mantenha como esta
   VideoRam        4096               #Memoria da placa. Normalmente esse e
o padrao


EndSection
